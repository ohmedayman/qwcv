    <!-- ===== SCRIPTS ===== -->
    <script>
(function() {
    var DB = 'https://qwcv-1cfad-default-rtdb.firebaseio.com';

    // ===== REST API =====
    function rGet(p){return fetch(DB+p+'.json').then(function(r){return r.json()}).catch(function(){return null})}
    function rPut(p,d){return fetch(DB+p+'.json',{method:'PUT',headers:{'Content-Type':'application/json'},body:JSON.stringify(d)}).catch(function(){})}
    function rPatch(p,d){return fetch(DB+p+'.json',{method:'PATCH',headers:{'Content-Type':'application/json'},body:JSON.stringify(d)}).catch(function(){})}
    function rDel(p){return fetch(DB+p+'.json',{method:'DELETE'}).catch(function(){});}
    function rPost(p,d){return fetch(DB+p+'.json',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify(d)}).then(function(r){return r.json()}).catch(function(){return null})}

    // ===== STATE =====
    var allUsers={},allSubs={},allJobs={},allDomains={},allStaff={},allReferrals={},allCampaigns={},allCommissions={},allSettings={},allMessages={};
    var currentSubFilter='all',activityLog=[],PAGE_SIZE=15,currentPage={};

    // ===== HELPERS =====
    function esc(s){if(!s)return'';var d=document.createElement('div');d.textContent=String(s);return d.innerHTML}
    function fmtDate(ts){if(!ts)return'-';return new Date(ts).toLocaleDateString('ar-EG')}
    function fmtDateTime(ts){if(!ts)return'-';return new Date(ts).toLocaleString('ar-EG')}
    function planTxt(p){return p==='pro'?'احترافية':p==='unlimited'?'غير محدودة':'مجانية'}
    function planBadge(p){return p==='pro'?'badge-pro':p==='unlimited'?'badge-unlimited':'badge-free'}
    function roleTxt(r){return r==='super_admin'?'مدير عام':r==='admin'?'مدير':r==='manager'?'مدير قسم':'دعم فني'}
    function roleBadge(r){return r==='super_admin'?'badge-super_admin':r==='admin'?'badge-admin':r==='manager'?'badge-manager':'badge-support'}
    function avatarColor(uid){var c=['#0003c9','#059669','#8b5cf6','#f59e0b','#ef4444','#3b82f6'];return c[(uid||'u').charCodeAt(0)%c.length]}
    function timeAgo(ts){var d=Date.now()-ts;if(d<60000)return'الآن';if(d<3600000)return'منذ '+Math.floor(d/60000)+' دقيقة';if(d<86400000)return'منذ '+Math.floor(d/3600000)+' ساعة';return'منذ '+Math.floor(d/86400000)+' يوم'}
    function addActivity(type,text){activityLog.unshift({type:type,text:text,time:Date.now()});if(activityLog.length>50)activityLog.pop();renderActivityFeed()}

    // ===== TOAST =====
    function showToast(msg,type){var t=document.getElementById('toast');t.textContent=msg;t.className='toast toast-'+type;setTimeout(function(){t.classList.add('show')},10);setTimeout(function(){t.classList.remove('show')},3000)}
    window.closeModal=function(){document.getElementById('modal').classList.remove('show')};

    // ===== PANEL SWITCHING =====
    var panelTitles={dashboard:'لوحة التحكم',users:'إدارة المستخدمين',staff:'إدارة الموظفين',subscriptions:'الاشتراكات',affiliate:'التسويق بالعمولة',reports:'التقارير',analytics:'التحليلات',jobs:'الوظائف',domains:'النطاقات المعتمدة',settings:'الإعدادات',messages:'الرسائل'};
    window.switchPanel=function(panel){
        document.querySelectorAll('.panel').forEach(function(p){p.classList.remove('active')});
        document.querySelectorAll('.nav-item').forEach(function(n){n.classList.remove('active')});
        var t=document.getElementById('panel-'+panel);if(t)t.classList.add('active');
        var n=document.querySelector('.nav-item[data-panel="'+panel+'"]');if(n)n.classList.add('active');
        var h=document.getElementById('pageTitle');if(h)h.textContent=panelTitles[panel]||panel;
        if(panel==='reports')renderReports();
        if(panel==='analytics')loadAnalytics();
    };

    // ===== PAGINATION =====
    function getPage(arr,sec){var pg=currentPage[sec]||1;var s=(pg-1)*PAGE_SIZE;return{items:arr.slice(s,s+PAGE_SIZE),total:arr.length,page:pg,pages:Math.ceil(arr.length/PAGE_SIZE)}}
    function renderPag(sec,total){var pages=Math.ceil(total/PAGE_SIZE);if(pages<=1){var el=document.getElementById(sec+'-pagination');if(el)el.innerHTML='';return''}var pg=currentPage[sec]||1;var h='<div class="pagination">';h+='<button '+(pg<=1?'disabled':'')+' onclick="goPage(\''+sec+'\','+(pg-1)+')">السابق</button>';for(var i=1;i<=pages;i++){h+='<button class="'+(i===pg?'active':'')+'" onclick="goPage(\''+sec+'\','+i+')">'+i+'</button>'}h+='<button '+(pg>=pages?'disabled':'')+' onclick="goPage(\''+sec+'\','+(pg+1)+')">التالي</button>';h+='<span class="page-info">صفحة '+pg+' من '+pages+' ('+total+' عنصر)</span></div>';return h}
    window.goPage=function(sec,pg){currentPage[sec]=pg;
        if(sec==='users')renderUsersTable(allUsers);
        else if(sec==='subscriptions')renderSubscriptions();
        else if(sec==='staff')renderStaffTable(allStaff);
        else if(sec==='jobs')renderJobsTable(allJobs);
        else if(sec==='domains')renderDomainsTable(allDomains);
        else if(sec==='messages')renderMessagesTable(allMessages);
    };

    // ===== LOAD ALL DATA =====
    function initDashboard(){
        Promise.all([rGet('/users'),rGet('/subscriptions'),rGet('/jobs'),rGet('/domains'),rGet('/staff'),rGet('/referrals'),rGet('/affiliateCampaigns'),rGet('/affiliateCommissions'),rGet('/settings'),rGet('/messages')]).then(function(r){
            allUsers=r[0]||{};allSubs=r[1]||{};allJobs=r[2]||{};allDomains=r[3]||{};allStaff=r[4]||{};
            allReferrals=r[5]||{};allCampaigns=r[6]||{};allCommissions=r[7]||{};allSettings=r[8]||{};allMessages=r[9]||{};
            updateStats();renderUsersTable(allUsers);renderRecentUsers(allUsers);renderSubscriptions();renderStaffTable(allStaff);
            renderJobsTable(allJobs);renderDomainsTable(allDomains);renderMessagesTable(allMessages);renderActivityFeed();
            renderSettings();addActivity('info','تم تحميل لوحة التحكم');
        });
    }

    // ===== UPDATE STATS =====
    function updateStats(){
        var e=Object.entries(allUsers);
        document.getElementById('statUsers').textContent=e.length;
        document.getElementById('statActive').textContent=e.filter(function(x){return!x[1].banned}).length;
        document.getElementById('statPro').textContent=e.filter(function(x){return x[1].plan==='pro'||x[1].plan==='unlimited'}).length;
        document.getElementById('statStaff').textContent=Object.keys(allStaff).length;
        document.getElementById('statCampaigns').textContent=Object.keys(allCampaigns).length;
        var pending=Object.values(allSubs).filter(function(s){return s.status==='pending'}).length;
        document.getElementById('statPending').textContent=pending;
        var b=document.getElementById('pendingBadge');if(pending>0){b.textContent=pending;b.style.display='inline'}else{b.style.display='none'}
        var rev=0,pp=parseInt(allSettings.proPrice)||99,up=parseInt(allSettings.unlimitedPrice)||299;
        e.forEach(function(x){if(x[1].plan==='pro')rev+=pp;else if(x[1].plan==='unlimited')rev+=up});
        document.getElementById('statRevenue').textContent=rev;
        renderDashBarChart(e);
    }

    // ===== ACTIVITY FEED =====
    function renderActivityFeed(){
        var el=document.getElementById('activityFeed');
        if(!activityLog.length){el.innerHTML='<div style="text-align:center;color:var(--muted);padding:20px;font-size:0.85rem">لا توجد نشاطات بعد</div>';return}
        el.innerHTML=activityLog.slice(0,10).map(function(a){
            var ic='fas fa-info-circle',bg='rgba(0,3,201,0.1)',cl='var(--accent)';
            if(a.type==='success'){ic='fas fa-check';bg='rgba(5,150,105,0.1)';cl='var(--green)'}
            else if(a.type==='error'){ic='fas fa-times';bg='rgba(239,68,68,0.1)';cl='var(--red)'}
            else if(a.type==='warning'){ic='fas fa-exclamation';bg='rgba(245,158,11,0.1)';cl='var(--yellow)'}
            return'<div class="activity-item"><div class="activity-icon" style="background:'+bg+';color:'+cl+'"><i class="'+ic+'"></i></div><div class="activity-text">'+a.text+'<div class="activity-time">'+timeAgo(a.time)+'</div></div></div>';
        }).join('');
    }

    // ===== DASH BAR CHART =====
    function renderDashBarChart(entries){
        var el=document.getElementById('dashBarChart'),months=[],now=new Date();
        for(var i=5;i>=0;i--){var d=new Date(now.getFullYear(),now.getMonth()-i,1);months.push({label:d.toLocaleDateString('ar-EG',{month:'short'}),count:0,start:d.getTime(),end:new Date(d.getFullYear(),d.getMonth()+1,0).getTime()})}
        entries.forEach(function(e){if(e[1].registeredAt)months.forEach(function(m){if(e[1].registeredAt>=m.start&&e[1].registeredAt<=m.end)m.count++})});
        var mx=Math.max.apply(null,months.map(function(m){return m.count}).concat([1]));
        el.innerHTML=months.map(function(m){return'<div class="bar-row"><div class="bar-label">'+m.label+'</div><div class="bar-track"><div class="bar-fill blue" style="width:'+(m.count/mx*100)+'%">'+m.count+'</div></div></div>'}).join('');
    }

    // ===== USERS =====
    function renderUsersTable(data){
        var tb=document.getElementById('allUsers'),entries=Object.entries(data);
        if(!entries.length){tb.innerHTML='<tr><td colspan="6" class="empty-state"><p>لا يوجد مستخدمين</p></td></tr>';document.getElementById('users-pagination').innerHTML='';return}
        var pd=getPage(entries,'users');
        tb.innerHTML=pd.items.map(function(e){var uid=e[0],u=e[1],plan=u.plan||'free',c=avatarColor(uid);
            return'<tr><td><div class="user-cell"><div class="user-avatar" style="background:'+c+'">'+esc((u.name||'U')[0].toUpperCase())+'</div><div class="user-info"><div class="name">'+esc(u.name||'مستخدم')+'</div><div class="email">'+esc(u.email)+'</div></div></div></td><td><span class="badge '+planBadge(plan)+'">'+planTxt(plan)+'</span></td><td>'+fmtDate(u.registeredAt)+'</td><td>'+fmtDateTime(u.lastLogin)+'</td><td>'+(u.banned?'<span class="badge badge-banned">محظور</span>':'<span class="badge badge-approved">نشط</span>')+'</td><td><div class="actions"><button class="btn-action btn-view" onclick="viewUser(\''+uid+'\')"><i class="fas fa-eye"></i></button><button class="btn-action btn-change-plan" onclick="changePlan(\''+uid+'\')"><i class="fas fa-exchange-alt"></i></button>'+(u.banned?'<button class="btn-action btn-unban" onclick="toggleBan(\''+uid+'\',false)"><i class="fas fa-unlock"></i></button>':'<button class="btn-action btn-ban" onclick="toggleBan(\''+uid+'\',true)"><i class="fas fa-ban"></i></button>')+'<button class="btn-action btn-delete" onclick="deleteUser(\''+uid+'\')"><i class="fas fa-trash"></i></button></div></td></tr>';
        }).join('');
        document.getElementById('users-pagination').innerHTML=renderPag('users',pd.total);
    }
    function renderRecentUsers(data){
        var tb=document.getElementById('recentUsers'),entries=Object.entries(data).sort(function(a,b){return(b[1].lastLogin||0)-(a[1].lastLogin||0)}).slice(0,5);
        if(!entries.length){tb.innerHTML='<tr><td colspan="4" class="empty-state"><p>لا يوجد مستخدمين</p></td></tr>';return}
        tb.innerHTML=entries.map(function(e){var uid=e[0],u=e[1],plan=u.plan||'free',c=avatarColor(uid);
            return'<tr><td><div class="user-cell"><div class="user-avatar" style="background:'+c+'">'+esc((u.name||'U')[0].toUpperCase())+'</div><div class="user-info"><div class="name">'+esc(u.name||'مستخدم')+'</div><div class="email">'+esc(u.email)+'</div></div></div></td><td><span class="badge '+planBadge(plan)+'">'+planTxt(plan)+'</span></td><td>'+fmtDateTime(u.lastLogin)+'</td><td>'+(u.banned?'<span class="badge badge-banned">محظور</span>':'<span class="badge badge-approved">نشط</span>')+'</td></tr>';
        }).join('');
    }

        document.getElementById('modalConfirm').style.display='none';
        document.getElementById('modal').classList.add('show');
    };
    window.changePlan=function(uid){
        var u=allUsers[uid];if(!u)return;
        document.getElementById('modalTitle').textContent='تغيير باقة المستخدم';
        document.getElementById('modalBody').innerHTML='<p style="text-align:center;margin-bottom:15px">المستخدم: <strong>'+esc(u.name||u.email)+'</strong></p><label>الباقة الجديدة</label><select id="newPlan"><option value="free"'+(u.plan==='free'?' selected':'')+'>مجانية</option><option value="pro"'+(u.plan==='pro'?' selected':'')+'>احترافية</option><option value="unlimited"'+(u.plan==='unlimited'?' selected':'')+'>غير محدودة</option></select><label>المدة (أيام)</label><input type="text" id="planDuration" value="30">';
        document.getElementById('modalConfirm').style.display='block';
        document.getElementById('modalConfirm').textContent='تحديث';
        document.getElementById('modalConfirm').onclick=function(){
            var plan=document.getElementById('newPlan').value;
            var days=parseInt(document.getElementById('planDuration').value)||30;
            var expiry=plan==='free'?null:Date.now()+(days*86400000);
            rPatch('/users/'+uid,{plan:plan,planExpiry:expiry}).then(function(){
                allUsers[uid].plan=plan;allUsers[uid].planExpiry=expiry;
                addActivity('success','تغيير باقة '+u.name+' إلى '+planTxt(plan));
                showToast('تم تحديث الباقة','success');closeModal();renderUsersTable(allUsers);renderRecentUsers(allUsers);updateStats();
            });
        };
        document.getElementById('modal').classList.add('show');
    };
    window.toggleBan=function(uid,ban){
        rPatch('/users/'+uid,{banned:ban}).then(function(){
            allUsers[uid].banned=ban;
            addActivity('warning',(ban?'حظر':'فك حظر')+' المستخدم '+(allUsers[uid].name||allUsers[uid].email));
            showToast(ban?'تم حظر المستخدم':'تم فك الحظر',ban?'error':'success');
            renderUsersTable(allUsers);renderRecentUsers(allUsers);updateStats();
        });
    };
    window.deleteUser=function(uid){
        var u=allUsers[uid];
        document.getElementById('modalTitle').textContent='حذف المستخدم';
        document.getElementById('modalBody').innerHTML='<p style="text-align:center">هل أنت متأكد من حذف <strong>'+esc(u?.name||u?.email)+'</strong>؟ لا يمكن التراجع.</p>';
        document.getElementById('modalConfirm').style.display='block';
        document.getElementById('modalConfirm').textContent='حذف';
        document.getElementById('modalConfirm').onclick=function(){
            rDel('/users/'+uid).then(function(){
                delete allUsers[uid];addActivity('error','حذف المستخدم '+(u?.name||u?.email));
                showToast('تم الحذف','success');closeModal();renderUsersTable(allUsers);renderRecentUsers(allUsers);updateStats();
            });
        };
        document.getElementById('modal').classList.add('show');
    };
    window.filterUsers=function(){
        currentPage['users']=1;
        var q=(document.getElementById('searchUsers').value||'').toLowerCase();
        var f={};Object.entries(allUsers).forEach(function(e){if((e[1].name||'').toLowerCase().indexOf(q)>=0||(e[1].email||'').toLowerCase().indexOf(q)>=0)f[e[0]]=e[1]});
        renderUsersTable(f);
    };
    window.exportUsersCSV=function(){
        var entries=Object.entries(allUsers);if(!entries.length){showToast('لا يوجد مستخدمين','error');return}
        var h=['الاسم','البريد','الباقة','تاريخ التسجيل','آخر دخول','الحالة'];
        var rows=entries.map(function(e){var u=e[1];return[u.name||'',u.email||'',planTxt(u.plan||'free'),u.registeredAt?new Date(u.registeredAt).toLocaleString('ar-EG'):'',u.lastLogin?new Date(u.lastLogin).toLocaleString('ar-EG'):'',u.banned?'محظور':'نشط']});
        var csv='\uFEFF'+[h].concat(rows).map(function(r){return r.map(function(c){return'"'+String(c).replace(/"/g,'""')+'"'}).join(',')}).join('\n');
        var blob=new Blob([csv],{type:'text/csv;charset=utf-8;'});var url=URL.createObjectURL(blob);
        var a=document.createElement('a');a.href=url;a.download='qcv-users-'+new Date().toISOString().slice(0,10)+'.csv';a.click();URL.revokeObjectURL(url);
        showToast('تم التصدير','success');
    };

    // ===== SUBSCRIPTIONS =====
    function renderSubscriptions(){
        var tb=document.getElementById('subscriptionsList');
        var entries=Object.entries(allSubs).sort(function(a,b){return(b[1].createdAt||0)-(a[1].createdAt||0)});
        if(currentSubFilter!=='all')entries=entries.filter(function(e){return e[1].status===currentSubFilter});
        if(!entries.length){tb.innerHTML='<tr><td colspan="6" class="empty-state"><p>لا توجد طلبات</p></td></tr>';document.getElementById('subscriptions-pagination').innerHTML='';return}
        var pd=getPage(entries,'subscriptions');
        tb.innerHTML=pd.items.map(function(e){
            var id=e[0],s=e[1];
            var st=s.status==='pending'?'badge-pending':s.status==='approved'?'badge-approved':'badge-rejected';
            var stTxt=s.status==='pending'?'معلق':s.status==='approved'?'مقبول':'مرفوض';
            var c=avatarColor(s.userId);
            return'<tr><td><div class="user-cell"><div class="user-avatar" style="background:'+c+'">'+esc((s.userName||'U')[0].toUpperCase())+'</div><div class="user-info"><div class="name">'+esc(s.userName||'مستخدم')+'</div><div class="email">'+esc(s.userEmail)+'</div></div></div></td><td><span class="badge '+planBadge(s.plan)+'">'+planTxt(s.plan)+'</span></td><td>'+fmtDateTime(s.createdAt)+'</td><td>'+esc(s.paymentMethod||'-')+'</td><td><span class="badge '+st+'">'+stTxt+'</span></td><td><div class="actions">'+(s.status==='pending'?'<button class="btn-action btn-approve" onclick="approveSub(\''+id+'\')"><i class="fas fa-check"></i> موافقة</button><button class="btn-action btn-reject" onclick="rejectSub(\''+id+'\')"><i class="fas fa-times"></i> رفض</button>':'<button class="btn-action btn-view" onclick="viewSub(\''+id+'\')"><i class="fas fa-eye"></i></button>')+'<button class="btn-action btn-delete" onclick="deleteSub(\''+id+'\')"><i class="fas fa-trash"></i></button></div></td></tr>';
        }).join('');
        document.getElementById('subscriptions-pagination').innerHTML=renderPag('subscriptions',pd.total);
    }
    window.setSubFilter=function(f,btn){
        currentSubFilter=f;currentPage['subscriptions']=1;
        document.querySelectorAll('.filter-tab').forEach(function(t){t.classList.remove('active')});btn.classList.add('active');
        renderSubscriptions();
    };
    window.approveSub=function(id){
        var s=allSubs[id];if(!s)return;
        document.getElementById('modalTitle').textContent='تأكيد الموافقة';
        document.getElementById('modalBody').innerHTML='<p style="text-align:center">الموافقة على اشتراك <strong>'+esc(s.userName)+'</strong> - باقة '+planTxt(s.plan)+'؟</p>';
        document.getElementById('modalConfirm').style.display='block';document.getElementById('modalConfirm').textContent='موافقة';
        document.getElementById('modalConfirm').onclick=function(){
            var expiry=Date.now()+(30*86400000);
            rPatch('/subscriptions/'+id,{status:'approved',approvedAt:Date.now()}).then(function(){
                return rPatch('/users/'+s.userId,{plan:s.plan,planExpiry:expiry});
            }).then(function(){
                allSubs[id].status='approved';allSubs[id].approvedAt=Date.now();
                addActivity('success','موافقة على اشتراك '+s.userName+' - باقة '+planTxt(s.plan));
                showToast('تم قبول الاشتراك','success');closeModal();renderSubscriptions();updateStats();
            });
        };
        document.getElementById('modal').classList.add('show');
    };
    window.rejectSub=function(id){
        document.getElementById('modalTitle').textContent='رفض طلب الاشتراك';

        document.getElementById('modalConfirm').style.display='block';document.getElementById('modalConfirm').textContent='رفض';
        document.getElementById('modalConfirm').onclick=function(){

                allSubs[id].status='rejected';addActivity('error','رفض اشتراك '+(allSubs[id].userName||''));
                showToast('تم رفض الطلب','error');closeModal();renderSubscriptions();updateStats();
            });
        };
        document.getElementById('modal').classList.add('show');
    };
    window.deleteSub=function(id){
        var s=allSubs[id];
        document.getElementById('modalTitle').textContent='حذف الاشتراك';
        document.getElementById('modalBody').innerHTML='<p style="text-align:center">هل أنت متأكد من حذف اشتراك <strong>'+esc(s?.userName)+'</strong>؟</p>';

    function renderReports(){
        var entries=Object.entries(allUsers);
        var now=new Date(),months=[];
        for(var i=5;i>=0;i--){var d=new Date(now.getFullYear(),now.getMonth()-i,1);months.push({label:d.toLocaleDateString('ar-EG',{month:'short'}),count:0,start:d.getTime(),end:new Date(d.getFullYear(),d.getMonth()+1,0).getTime()})}
        entries.forEach(function(e){if(e[1].registeredAt)months.forEach(function(m){if(e[1].registeredAt>=m.start&&e[1].registeredAt<=m.end)m.count++})});
        var mx=Math.max.apply(null,months.map(function(m){return m.count}).concat([1]));
        document.getElementById('reportGrowth').innerHTML=months.map(function(m){return'<div class="bar-row"><div class="bar-label">'+m.label+'</div><div class="bar-track"><div class="bar-fill blue" style="width:'+(m.count/mx*100)+'%">'+m.count+'</div></div></div>'}).join('');
        var free=0,pro=0,unlim=0;entries.forEach(function(e){var p=e[1].plan||'free';if(p==='pro')pro++;else if(p==='unlimited')unlim++;else free++});
        var total=Math.max(free+pro+unlim,1);
        document.getElementById('reportDistribution').innerHTML='<div class="donut-chart"><div class="donut-visual" style="background:conic-gradient(#9ca3af 0% '+Math.round(free/total*100)+'%, var(--accent) '+Math.round(free/total*100)+'% '+Math.round((free+pro)/total*100)+'%, var(--purple) '+Math.round((free+pro)/total*100)+'% 100%)"><div class="donut-center">'+total+'</div></div><div class="donut-legend"><div class="legend-item"><div class="legend-dot" style="background:#9ca3af"></div> مجانية: '+free+'</div><div class="legend-item"><div class="legend-dot" style="background:var(--accent)"></div> احترافية: '+pro+'</div><div class="legend-item"><div class="legend-dot" style="background:var(--purple)"></div> غير محدودة: '+unlim+'</div></div></div>';
    }

    // ===== STAFF LOGIN =====
    document.getElementById('staffLoginForm').addEventListener('submit',function(e){
        e.preventDefault();
        var email=document.getElementById('staffLoginEmail').value.trim(),pass=document.getElementById('staffLoginPassword').value;
        var btn=document.getElementById('staffLoginBtn'),err=document.getElementById('staffLoginError');
        btn.textContent='جاري التحقق...';btn.disabled=true;err.style.display='none';
        rGet('/staff').then(function(staffData){
            staffData=staffData||{};var found=null,foundId=null;
            Object.entries(staffData).forEach(function(e){if(e[1].email===email&&e[1].password===pass){found=e[1];foundId=e[0]}});
            if(!found){err.textContent='بيانات الدخول غير صحيحة';err.style.display='block';btn.textContent='دخول';btn.disabled=false;return}
            if(found.active===false){err.textContent='هذا الحساب معطّل';err.style.display='block';btn.textContent='دخول';btn.disabled=false;return}
            var session={id:foundId,name:found.name,email:found.email,role:found.role,permissions:found.permissions||[]};
            sessionStorage.setItem('qcv_staff',JSON.stringify(session));
            document.getElementById('staffLogin').classList.add('hidden');
            document.getElementById('adminLayout').classList.add('show');
            initDashboard();
        }).catch(function(){err.textContent='خطأ في الاتصال';err.style.display='block';btn.textContent='دخول';btn.disabled=false});
    });
